SELECT COUNT(liv.editora) as quantidade, ed.nome , estado, cidade
FROM livro as liv
left join editora as ed
on liv.editora = ed.codeditora 
left join endereco as en
on ed.endereco = en.codendereco 
group by editora 
order by COUNT(liv.editora) desc
