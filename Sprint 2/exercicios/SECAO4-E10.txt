select ven.nmvdd as vendedor, sum((qtd * vrunt)) as valor_total_vendas,round(((select sum((qtd * vrunt))) * ven.perccomissao / 100 ),2) as comissao
from tbvendedor as ven
left join tbvendas as venda 
on ven.cdvdd = venda.cdvdd 
where status = 'Conclu√≠do'
group by venda.cdvdd 
order by sum((select (qtd * vrunt)) * ven.perccomissao / 100 ) desc
