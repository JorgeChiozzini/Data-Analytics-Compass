WITH QuantidadeVendas AS (
    SELECT
        ven.cdvdd,
        ven.nmvdd,
        COUNT(venda.cdvdd) AS quantidade_vendas
    FROM
        tbvendedor AS ven
    LEFT JOIN
        tbvendas AS venda ON ven.cdvdd = venda.cdvdd
    WHERE
        status = 'Conclu√≠do'
    GROUP BY
        ven.cdvdd, ven.nmvdd
)

SELECT
    cdvdd,
    nmvdd
FROM
    QuantidadeVendas
WHERE
    quantidade_vendas = (
        SELECT MAX(quantidade_vendas)
        FROM QuantidadeVendas
    )
