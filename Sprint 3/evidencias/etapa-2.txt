'''
Apresente a média de receita de bilheteria bruta dos principais 
filmes, considerando todos os atores. Estamos falando aqui da 
média da coluna Gross.
'''


with open('actors.csv', 'r') as file:
    lines = file.readlines()

header = lines[0]
lines = lines[1:]

last_values = [line.split(',')[-1] for line in lines]

mean = sum(float(value) for value in last_values) / len(last_values)

print(mean)


with open('mean.txt', 'w') as file:
  file.write(str(mean))
